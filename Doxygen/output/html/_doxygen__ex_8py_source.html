71def map_range(number, in_min, in_max, out_min, out_max):
72 """! Maps a number from one range to another.

74 @param number The input number to map.
75 @param in_min The minimum value of an input number.
76 @param in_max The maximum value of an input number.
77 @param out_min The minimum value of an output number.
78 @param out_max The maximum value of an output number.
79
80 @return The mapped number.
81 """
82
83 mapped = (number - in_min) * (out_max - out_min) / (in_max - in_min) + out_min
84 if out_min <= out_max:
